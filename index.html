var rightSide = document.querySelector('.right-side');
var leftSide = document.querySelector('.left-side');

leftSide.on('mouseenter, mouseleave', 'button, span, .img-div', function(e){
    e.stopPropagation();
    if(e.type == 'mouseleave'){
        this.style.background = '';
    }else{
        this.style.background = 'red';
    }
});

leftSide.on('click', 'button, span, .img-div', function(e){
    e.stopPropagation();
    rightSide.appendChild(this);
});

var center = document.querySelector('.center');
var centerHandler = document.querySelector('.center-handler');
centerHandler.on('click', 'button', function(){
    if(!leftSide.children.length && !rightSide.children.length){
        var len = center.children.length;
        for(var i = 0; i < len; i++) {
            if(i % 2 == 0) 
                rightSide.appendChild(center.children[0]);
            else
                leftSide.appendChild(center.children[0]);
            this.innerText = 'Pull top items to neutral area';
        };
    }else{
        if(leftSide.children.length)
            center.appendChild(leftSide.children[0]);
        if(rightSide.children.length)
            center.appendChild(rightSide.children[0]);
    }
    if(!leftSide.children.length && !rightSide.children.length)
        this.innerText = 'Divide the items from the containers';
});
